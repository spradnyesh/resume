\section{Academic Projects}
\subsection{Performance Appraisal in organization}
\cventry{Objective}{The main objectives of this project were to know how the employees of the organization are awarded on their performances and how it motivates them to perform better}{}{}{}{}
\cventry{Findings}{}{}{}{}{
\begin{itemize}
  \item The performance appraisal was not done on a regular basis
  \item All employees were evaluated on the same criteria and not based on their positions and jobs
  \item Lack of transparency in the evaluation process
\end{itemize}}
\cventry{Suggestions made}{}{}{}{}{
\begin{itemize}
  \item Performance appraisal has to be conducted regularly for it to be effective
  \item There should not be a common appraisal form for every job position in the organization
  \item The job and role expected from the employees should be decided well in advance and that too with the consensus with them
  \item System should be transparent, through the discussion about the performance with the concerned employee
\end{itemize}}
\subsection{Recruitment and Selection Process}
\cventry{Objective}{The aim of this project was to understand the internal recruitment process, to identify areas where there can be scope for improvement and to give a suitable recommendation to streamline the hiring process}{}{}{}{}
\cventry{Findings}{}{}{}{}{
\begin{itemize}
  \item Rigorous Pre-Hiring assessment tests to understand aptitude and personality of candidates
  \item Campus recruitments have huge potential for fulfilling manpower requirements cost effectively
  \item Lengthy pre-offer formalities
\end{itemize}}
\cventry{Suggestions made}{}{}{}{}{
\begin{itemize}
  \item Reduce the "waiting time" in your hiring process
  \item Tie up with more and more consultants from multiple segments
\end{itemize}}
